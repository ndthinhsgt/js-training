// 4. for while
// for => vòng lặp, sẽ lặp lại cho đến khi nào điều kiện không còn đúng nữa
// for (tạo biến, điều kiện chạy tiếp, thay đổi biến)
// nếu biến không thay đổi vòng lặp sẽ chạy mãi mãi => nên tránh
for (let i = 0; i < 10; i++) {
// vòng lặp chạy từ 0 đến 9 (không có 10 vì < 10)
if (i == 5) {
// nếu i bằng 5 thì bỏ qua vòng lặp này và chạy vòng lặp i = 6
continue;
}
if (i == 7) {
// nếu i bằng 7 thì dừng vòng lặp
break;
}
console.log("i", i);
}

// ví dụ bắt người dùng nhập email (có chứa @) => nếu không có @ thì người dùng sẽ phải nhập mãi mãi cho đến khi có @

// Lưu ý khi dùng for / while vòng lặp rất dễ dẫn đến vĩnh viễn gọi là infinite loop
// vì vậy trong các logic phức tạp hoặc không rõ nghĩa người ta sẽ tạo ra 1 giới hạn
// ví dụ như:
// ví dụ bắt người dùng nhập email (có chứa @) => nếu không có @ thì người dùng sẽ phải nhập mãi mãi cho đến khi có @
// nhưng nếu người dùng không bao giờ bấm @ thì sẽ không bao giờ dừng lại
// vì vậy người ta sẽ tạo ra 1 giới hạn như sau
let email = prompt("Nhập email của bạn: "); // thiếu @
let limit = 0;
const maxTry = 3; // tối đa 3 lần
while (!email.includes("@")) {
// giả sử người dùng nhập email mới
email = prompt("Vui lòng nhập email có chứa @: ");
limit++;
if (limit >= maxTry) {
alert("Xin lỗi, bạn đã nhập quá số lần cho phép, làm ơn thử lại sau");
break;
}
}
